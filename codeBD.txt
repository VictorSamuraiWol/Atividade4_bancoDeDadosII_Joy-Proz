CREATE TABLE cliente (
  id SERIAL primary key,
  nome VARCHAR(50)
);

CREATE TABLE cadastro (
	ID SERIAL PRIMARY KEY,
	qnt_cadastro INT
);

CREATE or REPLACE FUNCTION update_qnt_cadastro(x INT) 
	RETURNS INT AS $$
    BEGIN
    	UPDATE cadastro SET qnt_cadastro = x;
        RETURN qnt_cadastro FROM cadastro;
    END
$$ LANGUAGE PLPGSQL;
